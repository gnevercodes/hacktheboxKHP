# Base image
FROM ubuntu:latest

# Set working directory
WORKDIR /app

# Copy necessary files into the container
COPY . /app

# Install dependencies (if required)
RUN apt-get update && apt-get install -y \
    gdb \
    strace \
    ltrace \
    python3 \
    python3-pip && \
    apt-get clean

# Expose the port the server listens on
EXPOSE 1337

# Command to start the server
CMD ["./khp_server"]
